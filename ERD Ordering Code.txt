Table Customer {
  Customerid integer [primary key]
  FirstName varchar
  LastName varchar
  created_at timestamp
}

Table Customer_Address{
CustomerAddID interger [primary key]
Address1 varchar
Address2 varchar
Address3 varchar
Address4 varchar
PostCode varchar
DefaultFlag bit
}

Table Products {
  ProductID integer [primary key]
  Name varchar
  Description text [note: 'comment on product']
  Price Money
  created_at timestamp
}

Table Orders {
  OrderID interger [primary key]
  Customerid int
  DeliveryTime datetime
  OrderedDate datetime
}

Table Order_Items {
  OrderItemID interger [primary key]
  Quantity int
  ProductID int
}


Table Categories {
CategoryID interger [primary key]
CatType varchar
CatDescription text
}

Table Payment{
PaymentID INT [primary key]
PaymentType varchar

}

Table Transaction {
TransactionID INT [primary key]
OrderID interger
PaymentID INT
Amount Money
}

Ref Cus_Add: Customer_Address.CustomerAddID > Customer.Customerid // many-to-one

Ref Product_Cat: Products.ProductID > Categories.CategoryID 

Ref Ordering: Orders.OrderID > Order_Items.OrderItemID

Ref CusOrder: Orders.Customerid > Customer.Customerid

Ref CusOrder: Order_Items.ProductID > Products.ProductID

Ref Trans: Transaction.OrderID > Orders.OrderID 

Ref Pay: Transaction.PaymentID > Payment.PaymentID

Ref PayCus: Payment.PaymentID > Customer.Customerid
